trimmean_func <- function(x, weights=rep(1,length(x)),...) {
  meanest <- DescTools::MeanCI(x, trim=5/100, na.rm=TRUE, conf.level=0.95)
  as.tbstat(meanest, sep=" ", parens=c("(", ")"), sep2=", ")
}
